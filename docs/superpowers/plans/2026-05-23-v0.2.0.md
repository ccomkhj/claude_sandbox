def terminate_pid(pid: int, timeout_s: float = 5.0) -> None:
    """SIGTERM the process, wait up to timeout_s for it to exit, then SIGKILL.
    Tolerates ProcessLookupError (process already gone)."""
    try:
        os.kill(pid, _signal.SIGTERM)
    except ProcessLookupError:
        return
    deadline = time.monotonic() + timeout_s
    while time.monotonic() < deadline:
        try:
            os.kill(pid, 0)  # probe — does process still exist?
        except ProcessLookupError:
            return
        time.sleep(0.2)
    try:
        os.kill(pid, _signal.SIGKILL)
    except ProcessLookupError:
        pass
```

def _wait_for_db_ready(*, container: str, db_name: str, timeout_s: float = 60.0) -> None:
    import time as _time
    deadline = _time.time() + timeout_s
    while _time.time() < deadline:
        probe = docker.exec_in_container(
            container=container,
            cmd=["pg_isready", "-U", "postgres", "-d", db_name],
        )
        if probe.returncode == 0:
            return
        _time.sleep(1.0)
    raise RuntimeError(f"db container {container} never became ready")


def _import_dump_at_runtime(*, container: str, local_dump: Path, db_name: str) -> None:
    """Copy the dump into the running db container, restore it, then delete it."""
    docker.cp(src=str(local_dump), dst=f"{container}:/tmp/dump.dump")
    restore = docker.exec_in_container(
        container=container,
        cmd=[
            "pg_restore",
            "--username", "postgres",
            "--dbname", db_name,
            "--no-owner", "--no-acl",
            "--clean", "--if-exists",
            "--exit-on-error",
            "/tmp/dump.dump",
        ],
    )
    if restore.returncode != 0:
        raise RuntimeError(f"pg_restore failed: rc={restore.returncode}")
    # Best-effort wipe; tolerate `shred` not being present in the image.
    try:
        docker.exec_in_container(container=container, cmd=["shred", "-u", "/tmp/dump.dump"])
    except Exception:
        docker.exec_in_container(container=container, cmd=["rm", "-f", "/tmp/dump.dump"])
```

def _fmt_ts(epoch: Optional[float]) -> str:
    if epoch is None:
        return "-"
    return datetime.fromtimestamp(epoch, tz=timezone.utc).astimezone().strftime("%Y-%m-%d %H:%M:%S")


def cmd_list(args: argparse.Namespace) -> int:
    rows = session.all_sessions()
    if not rows:
        print("no sessions")
        return 0
    headers = ("ID", "STATUS", "BRANCH", "STARTED", "GOAL")
    print(f"{headers[0]:<26}  {headers[1]:<10}  {headers[2]:<30}  {headers[3]:<19}  {headers[4]}")
    for m in rows:
        goal_short = (m.goal[:40] + "…") if len(m.goal) > 40 else m.goal
        print(f"{m.id:<26}  {m.status:<10}  {m.branch:<30}  {_fmt_ts(m.started_at):<19}  {goal_short}")

    print(f"branch:   {meta.branch}")
    print(f"started:  {_fmt_ts(meta.started_at)}")
    print(f"finished: {_fmt_ts(meta.finished_at)}")

class DockerNotRunning(RuntimeError):
    """Raised when the Docker daemon is not reachable."""


def _run(argv: list[str], *, check: bool = True, capture: bool = True) -> subprocess.CompletedProcess:
    try:
        return subprocess.run(argv, check=check, capture_output=capture, text=True)
    except subprocess.CalledProcessError as e:
        # Surface daemon-down errors as a domain exception
        stderr = (e.stderr or "").lower()
        if any(s in stderr for s in (
            "cannot connect to the docker daemon",
            "is the docker daemon running",
            "daemon socket",
        )):
            raise DockerNotRunning(e.stderr.strip() if e.stderr else "Docker daemon not reachable") from e
        raise
